export const ContactContent = () => {
  const contactImagePath = "/images/contact/contact.jpg";

  const [form, setForm] = useState({
    fname: "",
    lname: "",
    email: "",
    message: "",
  });
  const [result, setResult] = useState("");
  const [loading, setLoading] = useState(false);

  const handleChange = (
    e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement>
  ) => {
    setForm({ ...form, [e.target.name]: e.target.value });
  };

  const handleSubmit = async (e: React.FormEvent) => {
    if (
      form.fname === "" ||
      form.lname === "" ||
      form.email === "" ||
      form.message === ""
    ) {
      setResult("Please fill all the fields");
      return;
    }

    e.preventDefault();
    setLoading(true);
    setResult("Sending...");

    const data = {
      access_key: "4a825a6a-442e-4605-a310-96c8ad3f0854",
      name: form.fname + " " + form.lname,
      email: form.email,
      message: form.message,
    };

    try {
      const response = await fetch("https://api.web3forms.com/submit", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Accept: "application/json",
        },
        body: JSON.stringify(data),
      });

      const resultData = await response.json();

      if (resultData.success) {
        setResult("Thank you for contacting us!");
        setForm({ fname: "", lname: "", email: "", message: "" });
      } else {
        setResult(
          resultData.message || "Something went wrong. Please try again."
        );
      }
    } catch (error) {
      setResult("An error occurred. Please try again.");
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="w-full ">
      <div className="w-full lg:h-[400px] bg-black h-[200px] flex items-center  flex-col  bg-gradient-to-bl from-[#1CA36E]/30 via-black to-[#1CA36E]/30    ">

          <h3>Get In Touch</h3>
        </div>
      </div>

      <div className="w-full p-5 lg:my-20 my-10">
        <div className="lg:w-[95%] w-full mx-auto flex items-center justify-between lg:flex-row flex-col">
          <div className="lg:w-[50%] w-full">
            <div className="w-full">
              <h3 className="text-3xl font-gilroy-bold">Let's Talk</h3>
              <p className="my-3 text-xl">
                Get in touch with us using the enquiry form or{" "}
                <br className="lg:block hidden" /> contact details below
              </p>
            </div>

            <div>
              {result && (
                <div className="w-full mt-5">
                  <p className="text-xl text-green-500">{result}</p>
                </div>
              )}
            </div>

            {/* form */}

            <div className="w-full mt-10">
              <form className="w-full" onSubmit={handleSubmit}>
                <div className="w-full flex items-center justify-between">
                  <div className="w-[45%]">
                    <label

                      First Name

                      onChange={handleChange}
                    />
                  </div>

            blog.image.startsWith("/") ? blog.image.substring(1) : blog.image,
            ../public/${blog.image.replace(/^\//, "")},
            ${window.location.origin}/${blog.image.replace(/^\//, "")},
          ];

          console.log("Attempting to load image from paths:", paths);

          const tryNextPath = (index: number) => {
            if (index >= paths.length) return;

            const img = new Image();
            img.onload = () => {
              console.log("Image loaded successfully:", paths[index]);
              setImageSrc(paths[index]);
            };
            img.onerror = () => tryNextPath(index + 1);
            img.src = paths[index];
          };

          tryNextPath(0);
        } catch (error) {
          console.error("Error in image loading:", error);
        }
      }, 200);
    }
  }, [blog]);

  if (!id || !blog) {
    return <div>Blog not found</div>;
  }

  return (
    <div className="w-full my-20 flex items-center justify-center">
